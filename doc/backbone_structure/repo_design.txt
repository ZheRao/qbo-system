individual pipeline structure 

    qbo_pl/
        __init__.py
        config.py           # constants & config access (paths, env keys, etc.)
        schemas.py          # Spark StructType + TypedDicts for PL rows
        extract.py          # API calls, raw JSON pull (later)
        flatten.py          # schemas, crawler, flatten_partition
        business_rules.py   # _extract_accnum, _identify_product, etc.
        pipeline.py         # optional orchestration wrapper
        config.py           # constants, paths, maybe env var helpers

        utils/
            __init__.py
            spark.py       # spark_session(), common options
            logging.py     # logging setup

                company_id: str
                fiscal_year: int
                fiscal_period: int

                account_name: str
                amount: float

            PL_SCHEMA: StructType = StructType([...])

            def crawl_node(json_level: dict, acc_info: tuple[str, str] = ()) -> Iterable[PLEntry]:
                ...

            def flatten_partition(iter_rows: Iterable[dict]) -> Iterable[PLEntry]:
                ...

            def flatten_pl_df(df_raw: DataFrame) -> DataFrame:
                """Entry point: raw JSON DF → typed, flattened PL DF."""
                return (
                    df_raw.rdd
                    .mapPartitions(flatten_partition)
                    .toDF(schema=PL_SCHEMA)
                )

    - example of pipeline.py 

            # run_pl_pipeline.py (or pl_job.py)

            from qbo_pl.extract_raw import fetch_pl_reports
            from qbo_pl.flatten_pl import flatten_pl_df
            from qbo_pl.business_rules import apply_business_rules
            from qbo_pl.utils.spark import get_spark

            def main():
                spark = get_spark("QBO PL Pipeline")

                # 1. Extract or load raw
                df_raw = spark.read.json("...bronze path...")

                # 2. Flatten
                df_flat = flatten_pl_df(df_raw)

                # 3. Business rules
                df_enriched = apply_business_rules(df_flat)

                # 4. Write out
                df_enriched.write.mode("overwrite").parquet("...silver path...")

            if __name__ == "__main__":
                main()
